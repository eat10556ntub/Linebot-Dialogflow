# 04 文字外的回覆


### 1. 貼圖 (packageId=1, 2, 3或4)

``` javascript
{
    "line":{
         "type": "sticker",
         "packageId": "1",
         "stickerId": "1"
    }
}
```



### 2. 圖片

``` javascript
{
    "line":{
        "type": "image",
        "originalContentUrl": "自己的圖片位址",
        "previewImageUrl": "自己的圖片位址"
    }
}
```



### 3. 聲音(duration是音檔長度ms)

``` javascript
{
    "line":{
        "type": "audio",
        "originalContentUrl": "自己的聲音檔位址",
        "duration": 60000
    }
}
```



### 4. 影片

```javascript
{
    "line":{
        "type": "video",
        "originalContentUrl": "自己的影片位址",
        "previewImageUrl": "自己的圖片位址"
    }
}
```



### 5. Google地圖

```javascript
{
    "line": {
        "type": "location",
        "title": "國立臺北商業大學",
        "address": "台北市濟南路一段321號",
        "latitude": 25.043268,
        "longitude": 121.525279
    }
}
```




### 6. Carousel圖片選單

```javascript
{
    "line": {
        "type": "template",
        "altText": "this is a image carousel template",
        "template": {
            "type": "image_carousel",
            "columns": [{
                "imageUrl": "自己的圖片位址",
                "action": {
                    "type": "message",
                    "label": "這是圖片1",
                    "text": "圖片1"
                }
            }, {
                "imageUrl": "自己的圖片位址",
                "action": {
                    "type": "message",
                    "label": "這是圖片2",
                    "text": "圖片2"
                }
            }]
        }
    }
}
```

### 9. 圖文選單

```javascript
{
    "line": {
        "type": "template",
        "altText": "This is a buttons template",
        "template": {
            "type": "buttons",
            "thumbnailImageUrl": "https://自己的圖片位址",
            "imageAspectRatio": "rectangle",
            "imageSize": "cover",
            "imageBackgroundColor": "#FFFFFF",
            "title": "Menu",
            "text": "Please select",
            "defaultAction": {
                "type": "uri",
                "label": "View detail",
                "uri": "http://example.com/page/123"
            },
            "actions": [{
                "type": "postback",
                "label": "Buy",
                "data": "action=buy&itemid=123"
            },
            {
                "type": "postback",
                "label": "Add to cart",
                "data": "action=add&itemid=123"
            },
            {
                "type": "uri",
                "label": "View detail",
                "uri": "http://example.com/page/123"
            }]
        }
    }
}
```
